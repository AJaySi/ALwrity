```python
    def parse_sitemap_content(sitemap_urls):
        """Parse XML sitemaps and extract content information"""
        import xml.etree.ElementTree as ET

        all_pages = []
        for sitemap_url in sitemap_urls:
            try:
                # Parse XML sitemap
                tree = ET.parse(fetch_sitemap(sitemap_url))
                root = tree.getroot()

                # Extract URLs and metadata
                for url in root.findall('.//{http://www.sitemaps.org/schemas/sitemap/0.9}url'):
                    page_data = {
                        'url': url.find('.//{http://www.sitemaps.org/schemas/sitemap/0.9}loc').text,
                        'lastmod': url.find('.//{http://www.sitemaps.org/schemas/sitemap/0.9}lastmod'),
                        'priority': url.find('.//{http://www.sitemaps.org/schemas/sitemap/0.9}priority'),
                    }
                    all_pages.append(page_data)
            except Exception as e:
                logger.warning(f"Failed to parse sitemap {sitemap_url}: {e}")

        return all_pages
```
